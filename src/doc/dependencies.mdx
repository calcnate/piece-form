# Dependencies

import { Playground, Props } from 'docz';
import { Form, Field } from '../index';

<Playground>
  <Form
    onSubmit={(values) => {
      console.log(values);
    }}
    initialValues={{ name: "", password: "" }}
  >
    <div>
      <label>username: </label>
      <Field name="name" validator={formFieldValues=>{if(!formFieldValues.name)return "需要填写用户名"}}>
        <input 
          placeholder="username" 
          type="text" 
        />
      </Field>
    </div>
    <div>
      <label>password: </label>
      <Field name="password" validator={formFieldValues=>{
            if(!formFieldValues.password)return "需要填写密码"
          }
        }>
        {({error,value,onChange,onBlur})=>(
            <span>
              <input 
                placeholder="password" 
                type="password" 
                onChange={onChange} 
                onBlur={onBlur}
              />
              <span style={{color:'red'}}>{error}</span>
            </span>
          )
        }
      </Field>
    </div>
    <div>
      <label>confirm: </label>
      <Field name="confirm" dependencies={['password']} validator={(formFieldValues)=>{
            const value=formFieldValues.confirm
            if(!value)return "需要填写密码"
            if(formFieldValues.password!==value){
              return "两次输入的密码不一致"
            }
          }
        } >
        {({error,value,onChange,onBlur})=>(
            <span>
              <input 
                placeholder="confirm" 
                type="password" 
                onChange={onChange} 
                onBlur={onBlur}
              />
              <span style={{color:'red'}}>{error}</span>
            </span>
          )
        }
      </Field>
    </div>

    <button type="submit">Submit</button>

  </Form>
</Playground>
